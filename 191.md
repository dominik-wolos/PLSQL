declare
	cursor c1 is SELECT DISTINCT(DATA_EGZAMIN) as data FROM EGZAMINY e ORDER BY DATA_EGZAMIN DESC;
    cursor c2(data DATE) is SELECT DISTINCT er.IMIE, er.NAZWISKO FROM EGZAMINATORZY er INNER JOIN EGZAMINY en
        ON er.ID_EGZAMINATOR = en.ID_EGZAMINATOR
        WHERE en.DATA_EGZAMIN = data;
begin
	for vc1 in c1 loop
  	exit when c1%rowcount > 3 ;
  	-- dbms_output.put_line(vc1.data) ; 
        for vc2 in c2(vc1.data) LOOP
    		dbms_output.put_line(vc2.IMIE || ' ' || vc2.NAZWISKO) ; 
        end loop;
	end loop ; 
end ;
