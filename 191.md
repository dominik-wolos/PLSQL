Kuba
```sql
DECLARE
    vdata DATE;
    CURSOR cegzaminy IS
      SELECT DISTINCT data_egzamin
      FROM   egzaminy
      ORDER  BY data_egzamin DESC ;
    cursor cegzaminatorzy (pDate date) is select egz.Imie, egz.Nazwisko, egz.ID_EGZAMINATOR 
                                        from EGZAMINATORZY egz
                                            INNER JOIN EGZAMINY e ON e.ID_EGZAMINATOR = egz.ID_EGZAMINATOR
                                        where e.DATA_EGZAMIN = pDate ;
BEGIN
    OPEN cegzaminy;

    IF cegzaminy%isopen THEN
      FOR i IN 1..3 LOOP
          FETCH cegzaminy INTO vdata;

          EXIT WHEN cegzaminy%NOTFOUND;

          for cegzaminator in cegzaminatorzy(vdata) loop
            DBMS_OUTPUT.PUT_LINE(vdata || ' ' || cegzaminator.imie || ' ' || cegzaminator.nazwisko || ' ' || cegzaminator.ID_EGZAMINATOR);
          end loop;
      END LOOP;

      CLOSE cegzaminy;
    END IF;
END;
```

```sql
DECLARE
    vdata DATE;
    CURSOR cegzaminy IS
      SELECT DISTINCT data_egzamin
      FROM   egzaminy
      ORDER  BY data_egzamin DESC 
      FETCH FIRST 3 ROWS ONLY ;
    cursor cegzaminatorzy (pDate date) is select DISTINCT egz.Imie, egz.Nazwisko, egz.ID_EGZAMINATOR 
                                        from EGZAMINATORZY egz
                                            INNER JOIN EGZAMINY e ON e.ID_EGZAMINATOR = egz.ID_EGZAMINATOR
                                        where e.DATA_EGZAMIN = pDate ;
BEGIN
    FOR vc1 in cegzaminy loop
    		for cegzaminator in cegzaminatorzy(vc1.data_egzamin) loop
            DBMS_OUTPUT.PUT_LINE(vdata || ' ' || cegzaminator.imie || ' ' || cegzaminator.nazwisko || ' ' || cegzaminator.ID_EGZAMINATOR);
          end loop;
      END LOOP ;
END;
```
