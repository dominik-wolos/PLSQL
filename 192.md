```sql

DECLARE
	CURSOR Cprzedmioty IS
    SELECT DISTINCT
        P.ID_PRZEDMIOT, NAZWA_PRZEDMIOT
    FROM
        EGZAMINY E INNER JOIN PRZEDMIOTY P
  	ON P.ID_PRZEDMIOT = E.ID_PRZEDMIOT ;
  
  cursor datyPrzedmiot(przedmiot INT) is 
      SELECT distinct e.DATA_EGZAMIN FROM EGZAMINY e WHERE e.ID_PRZEDMIOT = przedmiot 
          ORDER  BY data_egzamin DESC 
          FETCH FIRST 2 ROWS ONLY ;
          
      cursor cstudent (p_id_przedmiot INT, p_data DATE) is 
        select distinct s.id_student s.IMIE, s.NAZWISKO from studenci s 
        inner join EGZAMINY ex on ex.id_student= s.ID_STUDENT
            where ex.id_przedmiot = p_id_przedmiot
            and ex.data_egzamin = p_data
            order by s.nazwisko, s.imie;
BEGIN
	FOR VC1 IN Cprzedmioty LOOP
  	DBMS_OUTPUT.PUT_LINE(VC1.NAZWA_PRZEDMIOT) ; 
    for vc2 in datyPrzedmiot(vc1.id_przedmiot) loop
    	DBMS_OUTPUT.PUT_LINE(to_char(VC2.data_egzamin, 'yyyy-mm-dd') ; 
      
                  FOR VC3 IN cstudent(VC1.ID_PRZEDMIOT, VC2.DATA_EGZAMIN) LOOP
                DBMS_OUTPUT.PUT_LINE(VC3.NAZWISKO || ' ' || VC3.NAZWISKO || ' ' || VC3.IMIE );
            END LOOP; 

      
    end loop;
  END LOOP ; 
END ;
```
